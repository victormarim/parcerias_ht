    <script>
        // --- CONFIGURAÇÃO ---
        // !!! IMPORTANTE !!!
        // COLE A URL DO SEU GOOGLE APPS SCRIPT AQUI (a mesma de antes, ou uma nova se preferir)
        const GOOGLE_APP_SCRIPT_URL = 'https://script.google.com/macros/s/AKfycbwKS7QNAEv1yGpMNB4hEAx8M-omCucQT6qUa5LK-dgATo2HwugeSjwe_xcaFPJk3pOf/exec'; 
        // --------------------

        // Lógica para campos condicionais
        const mentoriaSim = document.getElementById('mentoria-sim');
        const mentoriaNao = document.getElementById('mentoria-nao');
        const mentoriaDetalhes = document.getElementById('mentoria-detalhes');

        mentoriaSim.addEventListener('change', () => {
            if (mentoriaSim.checked) {
                mentoriaDetalhes.classList.remove('hidden');
            }
        });

        mentoriaNao.addEventListener('change', () => {
            if (mentoriaNao.checked) {
                mentoriaDetalhes.classList.add('hidden');
                // Limpa os campos quando escondidos
                document.getElementById('qtd-alunos').value = '';
                document.getElementById('periodicidade-aulas').value = '';
            }
        });

        function showAlert(title, message) {
            const errorMsg = document.getElementById('error-message');
            document.getElementById('error-text').textContent = title;
            document.getElementById('error-subtext').textContent = message;
            errorMsg.classList.remove('hidden');
            document.getElementById('success-message').classList.add('hidden');
            window.scrollTo(0, 0); // Rola para o topo para ver a msg
            setTimeout(() => errorMsg.classList.add('hidden'), 5000);
        }

        document.getElementById('form-parceria').addEventListener('submit', function(event) {
            event.preventDefault();
            
            if (GOOGLE_APP_SCRIPT_URL === 'COLE_A_SUA_URL_DO_APPS_SCRIPT_AQUI') {
                showAlert('Configuração Incompleta', 'Por favor, adicione a URL do seu Google Apps Script no arquivo index.html.');
                return;
            }

            const form = event.target;
            const formData = new FormData(form);
            
            const payload = {
                nome: formData.get('nome'),
                email: formData.get('email'),
                crm: formData.get('crm'),
                trabalhaInjetaveis: formData.get('trabalha-injetaveis'),
                linhaTratamento: formData.get('linha-tratamento'),
                objetivoParceria: formData.get('objetivo-parceria'),
                compraOutroLab: formData.get('compra-outro-lab'),
                redesSociais: formData.get('redes-sociais'),
                realizaMentoria: formData.get('realiza-mentoria'),
                qtdAlunos: formData.get('qtd-alunos'),
                periodicidadeAulas: formData.get('periodicidade-aulas'),
                eSpeaker: formData.get('e-speaker')
            };

            // Validação simples
            if (!payload.nome || !payload.email || !payload.crm || !payload.trabalhaInjetaveis || !payload.objetivoParceria) {
                 showAlert('Campos Obrigatórios', 'Por favor, preencha todos os campos marcados com *.');
                 return;
            }

            const submitButton = document.getElementById('submit-button');
            const buttonText = document.getElementById('button-text');
            const loader = document.getElementById('loader');
            
            buttonText.textContent = 'Enviando...';
            loader.classList.remove('hidden');
            submitButton.disabled = true;
            document.getElementById('error-message').classList.add('hidden');
            document.getElementById('success-message').classList.add('hidden');

            fetch(GOOGLE_APP_SCRIPT_URL, {
                method: 'POST',
                mode: 'no-cors', // Importante para evitar erros de CORS com Apps Script
                headers: {
                    'Content-Type': 'application/json',
                },
                body: JSON.stringify(payload)
            })
            .then(() => {
                document.getElementById('success-message').classList.remove('hidden');
                form.reset();
                mentoriaDetalhes.classList.add('hidden'); // Esconde campos condicionais
                window.scrollTo(0, 0); // Rola para o topo
            })
            .catch(error => {
                console.error('Error:', error);
                showAlert('Erro no Envio', 'Ocorreu um erro ao enviar os dados. Verifique o console e a configuração do script.');
            })
            .finally(() => {
                buttonText.textContent = 'Enviar Solicitação';
                loader.classList.add('hidden');
                submitButton.disabled = false;
                setTimeout(() => {
                    document.getElementById('success-message').classList.add('hidden');
                }, 8000);
            });
        });
    </script>
